# RaporKolay Nginx Configuration
# SSL will be configured by Certbot later

# Kernel API Backend (running on port 13401)
upstream raporkolay_api {
    server 127.0.0.1:13401;
    keepalive 64;
}

# Kernel API - kernel.raporkolay.com
server {
    listen 80;
    server_name kernel.raporkolay.com;
    
    # Direct proxy to kernel API
    location / {
        proxy_pass http://raporkolay_api;
        proxy_http_version 1.1;
        proxy_set_header Upgrade $http_upgrade;
        proxy_set_header Connection 'upgrade';
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Proto $scheme;
        proxy_cache_bypass $http_upgrade;
        proxy_read_timeout 300s;
        proxy_connect_timeout 75s;
    }
}

# Admin Panel - admin.raporkolay.com
server {
    listen 80;
    server_name admin.raporkolay.com;
    
    root /apps/raporkolay/adminpanel;
    index index.html;
    
    # Gzip compression
    gzip on;
    gzip_vary on;
    gzip_min_length 1024;
    gzip_types text/plain text/css text/xml text/javascript application/javascript application/json application/xml+rss;
    
    # Static files
    location / {
        try_files $uri $uri/ /index.html;
        add_header Cache-Control "no-cache, must-revalidate";
    }
    
    # Assets with cache
    location ~* \.(js|css|png|jpg|jpeg|gif|svg|ico|woff|woff2|ttf|eot)$ {
        expires 1y;
        add_header Cache-Control "public, immutable";
    }

        proxy_set_header Connection 'upgrade';

        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Proto $scheme;
        proxy_cache_bypass $http_upgrade;
        proxy_read_timeout 300s;
        proxy_connect_timeout 75s;
    }
}

# Client PWA - app.raporkolay.com
server {
    listen 80;
    server_name app.raporkolay.com;
    
    root /apps/raporkolay/client;
    index index.html;
    
    # Gzip compression
    gzip on;
    gzip_vary on;
    gzip_min_length 1024;
    gzip_types text/plain text/css text/xml text/javascript application/javascript application/json application/xml+rss application/manifest+json;
    
    # Static files
    location / {
        try_files $uri $uri/ /index.html;
        add_header Cache-Control "no-cache, must-revalidate";
    }
    
    # PWA manifest
    location /manifest.webmanifest {
        add_header Cache-Control "public, max-age=3600";
        add_header Content-Type application/manifest+json;
    }
    
    # Service worker
    location /sw.js {
        add_header Cache-Control "no-cache, must-revalidate";
        add_header Content-Type application/javascript;
    }
    
    # Workbox files
    location ~* /workbox-.*\.js$ {
        add_header Cache-Control "no-cache, must-revalidate";
        add_header Content-Type application/javascript;
    }
    
    # Assets with cache
    location ~* \.(js|css|png|jpg|jpeg|gif|svg|ico|woff|woff2|ttf|eot)$ {
        expires 1y;
        add_header Cache-Control "public, immutable";

# Web Site - raporkolay.com
server {
    listen 80;
    server_name raporkolay.com;
    
    root /apps/raporkolay/web;
    index index.html;
    
    # Gzip compression
    gzip on;
    gzip_vary on;
    gzip_min_length 1024;
    gzip_types text/plain text/css text/xml text/javascript application/javascript application/json application/xml+rss;
    
    # Static files
    location / {
        try_files $uri $uri/ /index.html;
        add_header Cache-Control "no-cache, must-revalidate";
    }
    
    # Downloads folder
    location /downloads/ {
        alias /apps/raporkolay/web/downloads/;
        autoindex off;
    }
    
    # Assets with cache
    location ~* \.(js|css|png|jpg|jpeg|gif|svg|ico|woff|woff2|ttf|eot|pdf)$ {
        expires 1y;
        add_header Cache-Control "public, immutable";
    }
}

# Redirect www to non-www
server {
    listen 80;
    server_name www.raporkolay.com;
    return 301 http://raporkolay.com$request_uri;
}

# Installation instructions:
# 1. Copy this file to /etc/nginx/sites-available/raporkolay
# 2. Create symlink: sudo ln -s /etc/nginx/sites-available/raporkolay /etc/nginx/sites-enabled/
# 3. Test config: sudo nginx -t
# 4. Reload nginx: sudo systemctl reload nginx
# 5. Setup SSL with certbot:
#    sudo certbot --nginx -d raporkolay.com -d www.raporkolay.com -d admin.raporkolay.com -d app.raporkolay.com -d kernel.raporkolay.com
